2.script.js function validateForm() { var name = document.getElementById("name").value; var email = document.getElementById("email").value; var course = document.getElementById("course").value; var gender = document.querySelector('input[name="gender"]:checked'); var message = document.getElementById("message"); // Check if name is empty if (name.trim() == "") { message.textContent = "Please enter a name."; return false; } // Check if email has a valid format using a regular expression if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) { message.textContent = "Please enter a valid email address."; return false; } // Check if a course is selected from the dropdown if (course == "") { message.textContent = "Please select a course."; return false; } // Check if a gender is selected if (!gender) { message.textContent = "Please select your gender."; return false; } // If all validation checks pass message.style.color = "green"; // Change message color for success message.textContent = "Form submitted successfully!"; // To see the success message without the page reloading, we return false. // In a real application, you would return true to allow the form to submit. return false; }
